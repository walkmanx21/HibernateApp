CREATE TABLE Person
(
    id   INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR NOT NULL,
    age  int     NOT NULL
);

CREATE TABLE Items
(
    id INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    person_id INT REFERENCES PERSON(id),
    item_name VARCHAR(100) NOT NULL
);
